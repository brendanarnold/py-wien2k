>>> from wien2k.utils import extract_isoenergy_mesh
>>> import wien2k
>>> import numpy as np

Create an array that represents a single peak

>>> peak = np.array([[ 1., 0.,  0.,  0.,  0.],
...       [ 2., 1.,  0.,  0.,  1.],
...       [ 3., 2.,  0.,  0.,  0.]])
>>> km = wien2k.Kmesh(peak)
>>> extract_isoenergy_mesh(km, 0.5)
array([[ 0.5,  0. ,  0. ],
       [ 1.5,  0. ,  0. ]])
